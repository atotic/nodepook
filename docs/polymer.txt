Polymer cheat sheet

declaring:
<polymer-element name='my-component' class='asdf'>
every my-component has class='asdf'

<script>
  var myPrivate_ = { some: data }
  Polymer('my-component', {
    get priv() { return myPrivate_; }
  });
</script>
